,[Customer].[CustomerHierachy].[Bill To Customer].members on rows
FROM [Wwidw]

GO

, [Delivery Date].[Calendar].[Calendar Year].&[CY2015].children on 1
FROM [Wwidw]
WHERE [Item].[Color].&[Red];
GO
